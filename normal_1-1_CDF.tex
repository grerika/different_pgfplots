\def\gausscdf(#1)(#2)(#3){0.5*(1+(erf((#1-#2)/(#3*sqrt(2)))))}%
\pgfmathdeclarefunction{gauss}{3}{\pgfmathparse{1/(#3*sqrt(2*pi))*exp(-((#1-#2)^2)/(2*#3^2))}}

\begin{document} 
	\begin{tikzpicture}
		\begin{axis}[
			samples=100,
			domain=-4.5:4.5,

			xmax=4.5,
			xtick = {-4,...,4},
			ymin=-0.01,
			ymax=1.1,
			ytick = {0,0.2,...,1.0},
			axis x line=middle,
			axis y line=left,			
%			ylabel=$F(x)$,
%			grid=both,
		]
	
		  \addplot[very thick,smooth,blue] gnuplot{\gausscdf(x)(1)(1)};
		\end{axis}
