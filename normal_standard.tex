\documentclass[border=2mm]{standalone}
\usepackage{pgfplots}

\pgfmathdeclarefunction{gauss}{3}{%
  \pgfmathparse{1/(#3*sqrt(2*pi))*exp(-((#1-#2)^2)/(2*#3^2))}%
}

\begin{document}
	\begin{tikzpicture}
		\begin{axis}[
			samples=100,
			domain=-3.2:3.2,
%			xlabel=$x$,
%			xtick = {-3,...,3},
			ymin=0,
			ymax=0.5,
%			ytick = {0,0.1,...,0.5},
			axis x line=middle,
			axis y line=left,			
			grid=both
		]

			\addplot [very thick,cyan!50!black] {gauss(x, 0, 1)};
		\end{axis}
	\end{tikzpicture}
\end{document}
