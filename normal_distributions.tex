\documentclass[border=2mm]{standalone}
\usepackage{pgfplots}

\pgfmathdeclarefunction{gauss}{3}{%
  \pgfmathparse{1/(#3*sqrt(2*pi))*exp(-((#1-#2)^2)/(2*#3^2))}%
}

\begin{document}
	\begin{tikzpicture}
		\begin{axis}[
			domain=-6:5.7,
			samples=400,
			xmin=-6,
			xmax=6,
			xtick = {-5,...,5},
			ymin=0,
			ymax=1.6,
			ytick = {0,0.2,...,1.4},
			axis x line=middle,
			axis y line=left,
			grid=both,
			legend pos = north west,
			legend style={font=\small},
		]


 		 \addplot [very thick,violet!80!black] {gauss(x, -3, 1.4142)};%var 2.0
		 \addplot [very thick,red!80!black] {gauss(x, 0, 1)};
		 \addplot [very thick,orange!80!black] {gauss(x, 0, 2.236068)}; % var = 5
		 \addplot [very thick,yellow!80!black] {gauss(x, 0, 0.4472136)}; % var = 0.2
		 \addplot [very thick,green!80!black] {gauss(x, -2, 0.7071068)};%var 0.5
 		 \addplot [very thick,blue!80!black] {gauss(x, 3, .3162)};%var 0.1

		 \legend{ $\mathbf N(-3;2.0)$,$\mathbf N(0;1.0)$,$\mathbf N(0;5.0)$, $\mathbf N(0;0.2)$, 
		 	$\mathbf N(-2;0.5)$, $\mathbf N(3;0.1)$};
	
		\end{axis}
	\end{tikzpicture}
\end{document}
