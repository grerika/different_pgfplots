\documentclass[border=2mm]{standalone}
\usepackage{pgfplots}

\pgfmathdeclarefunction{gauss}{3}{%
  \pgfmathparse{1/(#3*sqrt(2*pi))*exp(-((#1-#2)^2)/(2*#3^2))}%
}

\begin{document}
	\begin{tikzpicture}
		\begin{axis}[
			domain=-5:5,
			samples=400,
			xmin = -5.5,
			xmax = 5.5,
			xtick = {-5,...,5},			 
			ymin = 0,
			ymax = 1.5,
			ytick = {0,0.2,...,1.4},  			 
%			axis on top,
			axis x line=middle,
			axis y line=left,			 
			grid=both
		]

			\addplot [very thick,red] {gauss(x, 0, 2)};
			\addplot [very thick,orange] {gauss(x, 0, 1.0)};
			\addplot [very thick,yellow] {gauss(x, 0, 0.8)}; 
			\addplot [very thick,green] {gauss(x, 0, 0.6)};
			\addplot [very thick,blue] {gauss(x, 0, 0.4)};
			\addplot [very thick,violet] {gauss(x, 0, 0.3)};
			\legend{$\sigma=2.0$,$\sigma=1.0$, $\sigma=0.8$, $\sigma=0.6$, $\sigma=0.4$, $\sigma=0.2$};
			%Rainbow colors: RED, ORANGE, YELLOW,GREEN,BLUE,INDIGO,VIOLET			
		\end{axis}
	\end{tikzpicture}
\end{document}
