\documentclass[border=2mm]{standalone}
\usepackage{pgfplots}

\pgfmathdeclarefunction{gauss}{3}{%
  \pgfmathparse{1/(#3*sqrt(2*pi))*exp(-((#1-#2)^2)/(2*#3^2))}%
}

\begin{document} 
	\begin{tikzpicture}
		\begin{axis}[
			samples=200,
			domain=-4.5:4.5,  
			xmin=-4.5, xmax=4.5,
			xtick = {-4,...,4},			
			ymin=0,	ymax=0.45,
			ytick = {0,0.1,...,0.5},  			
			axis x line=middle,
			axis y line=left,
			%grid=both,
		]

			\addplot [very thick,red!50!black] {gauss(x, 0, 1)};
			\draw [blue!50!black,thick,dashed] (axis cs:0.0, 0) -- (axis cs:0.0, 0.5); 
		\end{axis}
	\end{tikzpicture}
\end{document}
